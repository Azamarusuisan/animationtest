**JS**
```js
(() => {
  // Generate JSON-LD automatically from breadcrumb HTML
  function generateBreadcrumbSchema() {
    const breadcrumb = document.querySelector('.breadcrumb');
    if (!breadcrumb) return;

    const items = Array.from(breadcrumb.querySelectorAll('.breadcrumb-item'));
    const baseUrl = window.location.origin;

    const itemListElement = items.map((item, index) => {
      const link = item.querySelector('a');
      const span = item.querySelector('span');
      const text = link ? link.textContent.trim() : span.textContent.trim();
      const href = link ? link.getAttribute('href') : window.location.pathname;

      // Convert relative URLs to absolute
      const absoluteUrl = href.startsWith('http')
        ? href
        : new URL(href, baseUrl).href;

      return {
        "@type": "ListItem",
        "position": index + 1,
        "name": text,
        "item": absoluteUrl
      };
    });

    const schema = {
      "@context": "https://schema.org",
      "@type": "BreadcrumbList",
      "itemListElement": itemListElement
    };

    // Insert or update schema script
    let schemaScript = document.querySelector('script[type="application/ld+json"]');
    if (!schemaScript) {
      schemaScript = document.createElement('script');
      schemaScript.type = 'application/ld+json';
      document.head.appendChild(schemaScript);
    }
    schemaScript.textContent = JSON.stringify(schema, null, 2);
  }

  // Truncate breadcrumb for mobile (alternative JS approach)
  function handleResponsiveBreadcrumb() {
    const breadcrumb = document.querySelector('.breadcrumb');
    if (!breadcrumb) return;

    const items = breadcrumb.querySelectorAll('.breadcrumb-item');
    const isMobile = window.innerWidth <= 640;

    if (isMobile && items.length > 3) {
      // Show first, last, and second-to-last items
      items.forEach((item, index) => {
        if (index === 0 || index >= items.length - 2) {
          item.style.display = 'flex';
        } else if (index === 1) {
          // Add ellipsis to second item
          item.innerHTML = '<span>...</span>';
          item.style.display = 'flex';
        } else {
          item.style.display = 'none';
        }
      });
    } else {
      // Show all items on desktop
      items.forEach(item => {
        item.style.display = 'flex';
      });
    }
  }

  // Initialize
  generateBreadcrumbSchema();

  // Optional: Handle window resize for responsive behavior
  // window.addEventListener('resize', handleResponsiveBreadcrumb);
  // handleResponsiveBreadcrumb();
})();
```
