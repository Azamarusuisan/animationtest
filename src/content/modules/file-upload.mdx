  js={`
(() => {
  const dropzone = document.getElementById('dropzone');
  const fileInput = document.getElementById('file-input');
  const browseBtn = document.getElementById('browse-btn');
  const fileList = document.getElementById('file-list');
  const uploadBtn = document.getElementById('upload-btn');
  const clearBtn = document.getElementById('clear-btn');
  const message = document.getElementById('upload-message');

  let selectedFiles = [];
  const MAX_FILE_SIZE = 5 * 1024 * 1024;

  function formatFileSize(bytes) {
    if (bytes === 0) return '0 Bytes';
    const k = 1024;
    const sizes = ['Bytes', 'KB', 'MB'];
    const i = Math.floor(Math.log(bytes) / Math.log(k));
    return Math.round(bytes / Math.pow(k, i) * 100) / 100 + ' ' + sizes[i];
  }

  function validateFile(file) {
    if (file.size > MAX_FILE_SIZE) {
      return { valid: false, error: \`ファイルサイズが大きすぎます (最大 \${formatFileSize(MAX_FILE_SIZE)})\` };
    }
    return { valid: true };
  }

  function renderFileItem(fileData) {
    const item = document.createElement('div');
    item.className = 'file-item';
    if (fileData.error) item.classList.add('error');

    const preview = document.createElement('div');
    preview.className = 'file-preview';
    if (fileData.file.type.startsWith('image/')) {
      const img = document.createElement('img');
      img.src = URL.createObjectURL(fileData.file);
      preview.appendChild(img);
    } else {
      preview.innerHTML = '<svg viewBox="0 0 24 24"><path d="M13 2H6a2 2 0 0 0-2 2v16a2 2 0 0 0 2 2h12a2 2 0 0 0 2-2V9z"></path><polyline points="13 2 13 9 20 9"></polyline></svg>';
    }

    item.innerHTML = \`
      <div class="file-info">
        <p class="file-name">\${fileData.file.name}</p>
        <p class="file-size">\${formatFileSize(fileData.file.size)}</p>
        \${fileData.error ? \`<p class="file-error">\${fileData.error}</p>\` : '<div class="file-progress"><div class="file-progress-bar"></div></div>'}
      </div>
      <button type="button" class="file-remove">
        <svg viewBox="0 0 24 24"><polyline points="3 6 5 6 21 6"></polyline><path d="M19 6v14a2 2 0 0 1-2 2H7a2 2 0 0 1-2-2V6m3 0V4a2 2 0 0 1 2-2h4a2 2 0 0 1 2 2v2"></path></svg>
      </button>
    \`;
    item.insertBefore(preview, item.firstChild);

    item.querySelector('.file-remove').addEventListener('click', () => {
      const index = selectedFiles.indexOf(fileData);
      if (index > -1) {
        selectedFiles.splice(index, 1);
        renderFileList();
      }
    });

    return item;
  }

  function addFiles(files) {
    const newFiles = Array.from(files).map(file => {
      const validation = validateFile(file);
      return {
        file: file,
        error: validation.valid ? null : validation.error,
        progress: 0
      };
    });

    selectedFiles.push(...newFiles);
    renderFileList();
  }

  function renderFileList() {
    fileList.innerHTML = '';
    selectedFiles.forEach(fileData => {
      fileList.appendChild(renderFileItem(fileData));
    });
    uploadBtn.disabled = selectedFiles.length === 0 || selectedFiles.every(f => f.error);
    clearBtn.disabled = selectedFiles.length === 0;
  }

  function showMessage(text, type) {
    message.textContent = text;
    message.className = 'upload-message show ' + type;
    setTimeout(() => message.classList.remove('show'), 5000);
  }

  async function uploadFiles() {
    const validFiles = selectedFiles.filter(f => !f.error);
    if (validFiles.length === 0) return;

    uploadBtn.disabled = true;

    for (const fileData of validFiles) {
      const item = Array.from(fileList.children).find(el =>
        el.querySelector('.file-name').textContent === fileData.file.name
      );
      const progressBar = item?.querySelector('.file-progress-bar');

      if (progressBar) {
        await new Promise(resolve => {
          let progress = 0;
          const interval = setInterval(() => {
            progress += Math.random() * 30;
            if (progress >= 100) {
              progress = 100;
              clearInterval(interval);
              resolve();
            }
            progressBar.style.width = progress + '%';
          }, 200);
        });
      }
    }

    showMessage(\`\${validFiles.length}個のファイルをアップロードしました\`, 'success');
    setTimeout(() => {
      selectedFiles = [];
      renderFileList();
    }, 2000);
  }

  dropzone.addEventListener('click', () => fileInput.click());
  browseBtn.addEventListener('click', (e) => {
    e.stopPropagation();
    fileInput.click();
  });

  fileInput.addEventListener('change', (e) => {
    if (e.target.files.length > 0) {
      addFiles(e.target.files);
      fileInput.value = '';
    }
  });

  ['dragenter', 'dragover', 'dragleave', 'drop'].forEach(eventName => {
    dropzone.addEventListener(eventName, (e) => {
      e.preventDefault();
      e.stopPropagation();
    });
  });

  ['dragenter', 'dragover'].forEach(eventName => {
    dropzone.addEventListener(eventName, () => dropzone.classList.add('drag-over'));
  });

  ['dragleave', 'drop'].forEach(eventName => {
    dropzone.addEventListener(eventName, () => dropzone.classList.remove('drag-over'));
  });

  dropzone.addEventListener('drop', (e) => {
    const files = e.dataTransfer.files;
    if (files.length > 0) addFiles(files);
  });

  uploadBtn.addEventListener('click', uploadFiles);
  clearBtn.addEventListener('click', () => {
    selectedFiles = [];
    renderFileList();
  });
})();
  `}
/>

**JS**
```js
(() => {
  const dropzone = document.getElementById('dropzone');
  const fileInput = document.getElementById('file-input');
  const browseBtn = document.getElementById('browse-btn');
  const fileList = document.getElementById('file-list');
  const uploadBtn = document.getElementById('upload-btn');
  const clearBtn = document.getElementById('clear-btn');
  const message = document.getElementById('upload-message');

  if (!dropzone || !fileInput) return;

  let selectedFiles = [];
  const MAX_FILE_SIZE = 10 * 1024 * 1024; // 10MB
  const ALLOWED_TYPES = ['image/jpeg', 'image/png', 'image/gif', 'image/webp', 'application/pdf'];

  // Utility functions
  function formatFileSize(bytes) {
    if (bytes === 0) return '0 Bytes';
    const k = 1024;
    const sizes = ['Bytes', 'KB', 'MB', 'GB'];
    const i = Math.floor(Math.log(bytes) / Math.log(k));
    return Math.round(bytes / Math.pow(k, i) * 100) / 100 + ' ' + sizes[i];
  }

  function validateFile(file) {
    if (!ALLOWED_TYPES.includes(file.type)) {
      return { valid: false, error: 'File type not supported' };
    }
    if (file.size > MAX_FILE_SIZE) {
      return { valid: false, error: `File too large (max ${formatFileSize(MAX_FILE_SIZE)})` };
    }
    return { valid: true };
  }

  function generateId() {
    return Date.now().toString(36) + Math.random().toString(36).substr(2);
  }

  function createFilePreview(file) {
    const preview = document.createElement('div');
    preview.className = 'file-preview';

    if (file.type.startsWith('image/')) {
      const img = document.createElement('img');
      img.src = URL.createObjectURL(file);
      img.alt = file.name;
      preview.appendChild(img);
    } else {
      preview.innerHTML = `
        <svg viewBox="0 0 24 24">
          <path d="M13 2H6a2 2 0 0 0-2 2v16a2 2 0 0 0 2 2h12a2 2 0 0 0 2-2V9z"></path>
          <polyline points="13 2 13 9 20 9"></polyline>
        </svg>
      `;
    }

    return preview;
  }

  function renderFileItem(fileData) {
    const item = document.createElement('div');
    item.className = 'file-item';
    item.dataset.id = fileData.id;
    if (fileData.error) {
      item.classList.add('error');
    }

    const preview = createFilePreview(fileData.file);

    item.innerHTML = `
      <div class="file-info">
        <p class="file-name">${fileData.file.name}</p>
        <p class="file-size">${formatFileSize(fileData.file.size)}</p>
        ${fileData.error ? `<p class="file-error">${fileData.error}</p>` : ''}
        ${!fileData.error ? '<div class="file-progress"><div class="file-progress-bar"></div></div>' : ''}
      </div>
      <div class="file-actions">
        <button type="button" class="file-remove" aria-label="Remove ${fileData.file.name}">
          <svg viewBox="0 0 24 24">
            <polyline points="3 6 5 6 21 6"></polyline>
            <path d="M19 6v14a2 2 0 0 1-2 2H7a2 2 0 0 1-2-2V6m3 0V4a2 2 0 0 1 2-2h4a2 2 0 0 1 2 2v2"></path>
          </svg>
        </button>
      </div>
    `;

    item.insertBefore(preview, item.firstChild);

    const removeBtn = item.querySelector('.file-remove');
    removeBtn.addEventListener('click', () => removeFile(fileData.id));

    return item;
  }

  function addFiles(files) {
    const newFiles = Array.from(files).map(file => {
      const validation = validateFile(file);
      return {
        id: generateId(),
        file: file,
        error: validation.valid ? null : validation.error,
        progress: 0
      };
    });

    selectedFiles.push(...newFiles);
    renderFileList();
    updateButtons();
  }

  function removeFile(id) {
    selectedFiles = selectedFiles.filter(f => f.id !== id);
    renderFileList();
    updateButtons();
  }

  function renderFileList() {
    fileList.innerHTML = '';
    selectedFiles.forEach(fileData => {
      const item = renderFileItem(fileData);
      fileList.appendChild(item);
    });
  }

  function updateButtons() {
    const hasValidFiles = selectedFiles.some(f => !f.error);
    uploadBtn.disabled = !hasValidFiles;
    clearBtn.disabled = selectedFiles.length === 0;
  }

  function showMessage(text, type) {
    message.textContent = text;
    message.className = 'upload-message show ' + type;
    setTimeout(() => {
      message.classList.remove('show');
    }, 5000);
  }

  function clearAll() {
    selectedFiles = [];
    fileList.innerHTML = '';
    updateButtons();
    message.classList.remove('show');
  }

  // Simulate upload
  async function uploadFile(fileData) {
    const item = fileList.querySelector(`[data-id="${fileData.id}"]`);
    if (!item) return;

    const progressBar = item.querySelector('.file-progress-bar');
    item.classList.add('uploading');

    return new Promise((resolve) => {
      let progress = 0;
      const interval = setInterval(() => {
        progress += Math.random() * 30;
        if (progress >= 100) {
          progress = 100;
          clearInterval(interval);
          item.classList.remove('uploading');
          resolve();
        }
        progressBar.style.width = progress + '%';
      }, 300);
    });
  }

  async function uploadFiles() {
    const validFiles = selectedFiles.filter(f => !f.error);
    if (validFiles.length === 0) return;

    uploadBtn.disabled = true;
    clearBtn.disabled = true;

    try {
      for (const fileData of validFiles) {
        await uploadFile(fileData);
      }
      showMessage(`Successfully uploaded ${validFiles.length} file(s)`, 'success');
      setTimeout(() => {
        clearAll();
      }, 2000);
    } catch (error) {
      showMessage('Upload failed. Please try again.', 'error');
      uploadBtn.disabled = false;
      clearBtn.disabled = false;
    }
  }

  // Event listeners - Dropzone
  dropzone.addEventListener('click', () => fileInput.click());

  dropzone.addEventListener('keydown', (e) => {
    if (e.key === 'Enter' || e.key === ' ') {
      e.preventDefault();
      fileInput.click();
    }
  });

  browseBtn.addEventListener('click', (e) => {
    e.stopPropagation();
    fileInput.click();
  });

  fileInput.addEventListener('change', (e) => {
    if (e.target.files.length > 0) {
      addFiles(e.target.files);
      fileInput.value = '';
    }
  });

  // Drag and drop
  ['dragenter', 'dragover', 'dragleave', 'drop'].forEach(eventName => {
    dropzone.addEventListener(eventName, (e) => {
      e.preventDefault();
      e.stopPropagation();
    });
  });

  ['dragenter', 'dragover'].forEach(eventName => {
    dropzone.addEventListener(eventName, () => {
      dropzone.classList.add('drag-over');
    });
  });

  ['dragleave', 'drop'].forEach(eventName => {
    dropzone.addEventListener(eventName, () => {
      dropzone.classList.remove('drag-over');
    });
  });

  dropzone.addEventListener('drop', (e) => {
    const files = e.dataTransfer.files;
    if (files.length > 0) {
      addFiles(files);
    }
  });

  // Buttons
  uploadBtn.addEventListener('click', uploadFiles);
  clearBtn.addEventListener('click', clearAll);

  // Prevent default drag behavior on document
  ['dragenter', 'dragover', 'drop'].forEach(eventName => {
    document.addEventListener(eventName, (e) => {
      if (!dropzone.contains(e.target)) {
        e.preventDefault();
      }
    }, false);
  });
})();
```
