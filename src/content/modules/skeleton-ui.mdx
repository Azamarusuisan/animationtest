**JS**
```js
(() => {
  // スケルトンUIを表示し、コンテンツ読み込み後に置き換える
  window.showSkeleton = (container, skeletonType = 'card') => {
    const skeletonTemplates = {
      card: `
        <div class="skeleton-card">
          <div class="skeleton skeleton-image"></div>
          <div class="skeleton-body">
            <div class="skeleton skeleton-title"></div>
            <div class="skeleton skeleton-text"></div>
            <div class="skeleton skeleton-text"></div>
            <div class="skeleton skeleton-text short"></div>
          </div>
        </div>
      `,
      list: `
        <div class="skeleton-list-item">
          <div class="skeleton skeleton-avatar"></div>
          <div class="skeleton-list-content">
            <div class="skeleton skeleton-text"></div>
            <div class="skeleton skeleton-text short"></div>
          </div>
        </div>
      `,
      lines: `
        <div class="skeleton-lines">
          <div class="skeleton skeleton-text"></div>
          <div class="skeleton skeleton-text"></div>
          <div class="skeleton skeleton-text short"></div>
        </div>
      `
    };

    const template = skeletonTemplates[skeletonType] || skeletonTemplates.card;
    container.innerHTML = template;
  };

  // コンテンツを読み込んでスケルトンを置き換える
  window.loadContent = async (container, fetchFn) => {
    try {
      const content = await fetchFn();

      // フェードアウト
      container.style.opacity = '0';

      setTimeout(() => {
        container.innerHTML = content;

        // フェードイン
        requestAnimationFrame(() => {
          container.style.opacity = '1';
        });
      }, 300);

    } catch (error) {
      console.error('コンテンツの読み込みに失敗:', error);
      container.innerHTML = '<p>読み込みに失敗しました。</p>';
    }
  };

  // 使用例:
  // const container = document.getElementById('content');
  // showSkeleton(container, 'card');
  // loadContent(container, async () => {
  //   const response = await fetch('/api/content');
  //   return await response.text();
  // });

  // 複数のスケルトンを一度に表示
  window.showMultipleSkeletons = (container, skeletonType = 'list', count = 3) => {
    const skeletons = Array.from({ length: count }, () => {
      const div = document.createElement('div');
      showSkeleton(div, skeletonType);
      return div.innerHTML;
    }).join('');

    container.innerHTML = skeletons;
  };
})();
```
