**JS**
```js
(() => {
  const forms = document.querySelectorAll('.validated-form');
  const msgs = {
    valueMissing: 'This field is required',
    typeMismatch: 'Please enter a valid format',
    tooShort: 'Too short (min {min} chars)',
    tooLong: 'Too long (max {max} chars)',
    patternMismatch: 'Invalid format'
  };

  function validate(input) {
    const error = input.parentElement.querySelector('.error');
    if (input.validity.valid) {
      input.setAttribute('aria-invalid', 'false');
      error.textContent = '';
      return true;
    }
    input.setAttribute('aria-invalid', 'true');
    for (let key in msgs) {
      if (input.validity[key]) {
        error.textContent = msgs[key]
          .replace('{min}', input.minLength || input.min)
          .replace('{max}', input.maxLength || input.max);
        return false;
      }
    }
    error.textContent = input.validationMessage;
    return false;
  }

  forms.forEach(form => {
    const inputs = form.querySelectorAll('input, textarea, select');
    inputs.forEach(input => {
      input.addEventListener('blur', () => validate(input));
      input.addEventListener('input', () => {
        if (input.getAttribute('aria-invalid') === 'true') validate(input);
      });
    });

    form.addEventListener('submit', (e) => {
      e.preventDefault();
      let first = null;
      inputs.forEach(input => {
        if (!validate(input) && !first) first = input;
      });
      if (first) {
        first.focus();
      } else {
        console.log('Form valid!', Object.fromEntries(new FormData(form)));
      }
    });
  });
})();
```
