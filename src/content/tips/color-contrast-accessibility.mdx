    function getLuminance(r, g, b) {
      const [rs, gs, bs] = [r, g, b].map(c => {
        c = c / 255;
        return c <= 0.03928 ? c / 12.92 : Math.pow((c + 0.055) / 1.055, 2.4);

## カラーコントラストとは

**コントラスト比**は、前景色（テキスト）と背景色の明るさの差を数値化したものです。視覚障害のあるユーザーや、明るい屋外での使用、高齢のユーザーにとって、適切なコントラストは読みやすさの鍵となります。

### コントラスト比の範囲

```
1:1    最小（白と白、黒と黒）
21:1   最大（黒と白）
```

### WCAG 2.1 基準

**レベルAA（推奨）:**
- 通常のテキスト: **4.5:1** 以上
- 大きなテキスト: **3:1** 以上

**レベルAAA（理想）:**
- 通常のテキスト: **7:1** 以上
- 大きなテキスト: **4.5:1** 以上

**大きなテキストの定義:**
- 18pt（24px）以上
- または 14pt（18.66px）の太字以上

## コントラスト比の計算

### 手動計算（JavaScript）

```javascript
// RGB値から相対輝度を計算
function getLuminance(r, g, b) {
  // RGB値を0-1に正規化
  const [rs, gs, bs] = [r, g, b].map(c => {
    c = c / 255;
    // sRGB色空間の変換
    return c <= 0.03928
      ? c / 12.92
      : Math.pow((c + 0.055) / 1.055, 2.4);
  });

  // ITU-R BT.709の係数
  return 0.2126 * rs + 0.7152 * gs + 0.0722 * bs;
}

// コントラスト比を計算
function getContrastRatio(rgb1, rgb2) {
  const lum1 = getLuminance(...rgb1);
  const lum2 = getLuminance(...rgb2);

  const lighter = Math.max(lum1, lum2);
  const darker = Math.min(lum1, lum2);

  return (lighter + 0.05) / (darker + 0.05);
}

// Hex色をRGBに変換
function hexToRgb(hex) {
  const result = /^#?([a-f\d]{2})([a-f\d]{2})([a-f\d]{2})$/i.exec(hex);
  return result ? [
    parseInt(result[1], 16),
    parseInt(result[2], 16),
    parseInt(result[3], 16)
  ] : null;
}

// 使用例
const bg = hexToRgb('#0c1218');
const fg = hexToRgb('#e5edf4');
const ratio = getContrastRatio(bg, fg);

console.log(`コントラスト比: ${ratio.toFixed(2)}:1`);

// WCAG基準をチェック
function checkWCAG(ratio) {
  return {
    normalAA: ratio >= 4.5,
    normalAAA: ratio >= 7,
    largeAA: ratio >= 3,
    largeAAA: ratio >= 4.5
  };
}
```

## よくある問題と解決策

### 1. リンクのコントラスト

```html
<!-- ❌ 悪い例: 背景とのコントラストが低い -->
<p style="color: #6b7280;">
  詳細は<a href="#" style="color: #9ca3af;">こちら</a>をご覧ください
</p>

<!-- ✅ 良い例: 十分なコントラスト -->
<p style="color: #6b7280;">
  詳細は<a href="#" style="color: #3b82f6;">こちら</a>をご覧ください
</p>
```

### 2. ボタンの状態

```css
/* ❌ 悪い例: disabled状態のコントラストが低すぎる */
.button:disabled {
  background: #d1d5db;
  color: #e5e7eb;
  /* コントラスト比: 1.4:1 */
}

/* ✅ 良い例: 最低限のコントラストを確保 */
.button:disabled {
  background: #9ca3af;
  color: #1f2a37;
  /* コントラスト比: 4.8:1 */
  opacity: 0.6; /* 視覚的に無効であることを示す */
}
```

### 3. プレースホルダーテキスト

```css
/* ❌ 悪い例: 読みにくいプレースホルダー */
input::placeholder {
  color: #d1d5db;
  /* コントラスト比: 1.8:1 */
}

/* ✅ 良い例: 読みやすいプレースホルダー */
input::placeholder {
  color: #6b7280;
  /* コントラスト比: 4.8:1 */
}
```

### 4. オーバーレイテキスト

```css
/* 画像上のテキストは背景を追加 */

```

## CSS変数で管理

```css
:root {
  /* カラーパレット */
  --color-bg-dark: #0c1218;
  --color-bg-medium: #1f2a37;
  --color-text-light: #e5edf4;
  --color-text-medium: #9ca3af;

  /* コントラストを保証したペア */
  --pair-dark-bg-light-text: var(--color-bg-dark) var(--color-text-light); /* 14.2:1 */
  --pair-medium-bg-light-text: var(--color-bg-medium) var(--color-text-light); /* 9.8:1 */
  --pair-medium-bg-medium-text: var(--color-bg-medium) var(--color-text-medium); /* 4.6:1 */
}

/* 使用例 */
.card {
  background: var(--color-bg-medium);
  color: var(--color-text-light);
  /* 自動的に 9.8:1 のコントラスト比を確保 */
}
```

## チェックリスト

アクセシブルなカラー設計:

- [ ] すべてのテキストでAAレベル以上を達成
- [ ] 重要なテキストでAAAレベルを目指す
- [ ] リンクは背景と4.5:1以上のコントラスト
- [ ] ボタンの全状態で十分なコントラスト
- [ ] プレースホルダーも読みやすい色
- [ ] 色だけでなくアイコンやラベルを併用
- [ ] ダークモードとライトモード両方で確認
- [ ] コントラストチェッカーでバリデーション
- [ ] 実機でテスト（屋外・夜間含む）

## まとめ

カラーコントラストは、アクセシビリティの基本です:

1. **WCAG基準を遵守** - AAレベル以上を目指す
2. **ツールを活用** - 自動チェックで効率化
3. **色だけに頼らない** - アイコンや形状も使用
4. **CSS変数で管理** - 一貫性のあるカラーシステム
5. **実機テスト** - 様々な環境で確認

適切なコントラストにより、すべてのユーザーが快適にコンテンツを読めるようになります。
