    const modal = document.getElementById('modal');
    const openModalBtn = document.getElementById('openModal');
    const closeModalBtn = document.getElementById('closeModal');
    let previousFocus;

    openModalBtn.addEventListener('click', () => {
      previousFocus = document.activeElement;
      modal.hidden = false;
      closeModalBtn.focus();

      // フォーカストラップの実装
      modal.addEventListener('keydown', trapFocus);

    function closeModal() {

    function trapFocus(e) {
      if (e.key === 'Escape') {
        closeModal();
        return;
      }

      if (e.key !== 'Tab') return;

      const focusableElements = modal.querySelectorAll(
        'button:not([disabled]), [href], input, select, textarea, [tabindex]:not([tabindex="-1"])'

      const firstElement = focusableElements[0];

        lastElement.focus();
        e.preventDefault();

## キーボードナビゲーションの重要性

多くのユーザーがマウスの代わりにキーボードでWebサイトを操作しています:

- **身体的障害のあるユーザー**: マウス操作が困難
- **視覚障害のあるユーザー**: スクリーンリーダーとキーボードを併用
- **パワーユーザー**: キーボード操作の方が効率的
- **一時的な制約**: 怪我、マウスの故障など

**すべてのインタラクティブ要素はキーボードでアクセス可能である必要があります。**

## 基本的なキーボード操作

### 標準のキーボードショートカット

```
Tab             次のフォーカス可能な要素へ移動
Shift + Tab     前のフォーカス可能な要素へ移動
Enter           リンクやボタンをアクティブ化
Space           ボタンをアクティブ化、チェックボックスをトグル
矢印キー        ラジオボタン、スライダー、カスタムウィジェット
Esc             モーダルやポップアップを閉じる
Home / End      コンテナの最初/最後の要素へ移動
```

### フォーカス可能な要素

デフォルトでフォーカス可能:
```html
<a href="...">リンク</a>
<button>ボタン</button>
<input type="text">
<select>...</select>
<textarea>...</textarea>
<button>送信</button>
```

フォーカス不可能（デフォルト）:
```html
<div>通常のdiv</div>
<span>通常のspan</span>
<p>段落</p>
```

## tabindex属性

`tabindex` は要素のフォーカス順序を制御します。

### tabindex="0"

要素を自然なTab順序に含める:

```html
<div tabindex="0" role="button">
  カスタムボタン（フォーカス可能）
</div>

<div class="card" tabindex="0">
  フォーカス可能なカード
</div>
```

### tabindex="-1"

プログラムからのみフォーカス可能:

```html
<div id="alert" tabindex="-1">
  エラーメッセージ
</div>

<script>
  // エラー発生時にフォーカス
  document.getElementById('alert').focus();
</script>
```

### tabindex="1+"（使用禁止）

```html
<!-- ❌ 絶対に使用しない -->
<input tabindex="1">
<button tabindex="2">
<a href="..." tabindex="3">

<!-- ✅ 自然な順序を維持 -->
<input>
<button>
<a href="...">
```

**理由:**
- HTML構造と異なる順序は混乱を招く
- メンテナンスが困難
- アクセシビリティの問題

## フォーカススタイルの実装

### 標準のフォーカススタイル

```css
/* デフォルトのアウトライン */
button:focus {
  outline: 2px solid #60a5fa;
  outline-offset: 2px;
}

/* より目立つスタイル */
button:focus-visible {
  outline: 3px solid #60a5fa;
  outline-offset: 3px;
  box-shadow: 0 0 0 4px rgba(96, 165, 250, 0.3);
}
```

### カスタムフォーカススタイル

```css
.button {
  position: relative;
  border: 2px solid transparent;
  transition: border-color 0.2s;
}

/* マウスクリックではフォーカスを表示しない */
.button:focus {
  outline: none;
}

/* キーボードフォーカスのみ表示 */
.button:focus-visible {
  border-color: #60a5fa;
  box-shadow: 0 0 0 4px rgba(96, 165, 250, 0.3);
}
```

### 絶対にやってはいけないこと

```css
/* ❌ フォーカスインジケーターを完全に削除 */
*:focus {
  outline: none; /* 絶対NG! */
}

button:focus {
  outline: none; /* 代替スタイルがなければNG! */
}
```

## フォーカストラップ（モーダル）

モーダル表示中、フォーカスをモーダル内に制限:

```javascript
function openModal(modal) {
  // 現在のフォーカス位置を保存
  const previousFocus = document.activeElement;

  // モーダルを表示
  modal.style.display = 'block';

  // モーダル内の最初の要素にフォーカス
  const firstFocusable = modal.querySelector('button, [href], input, select, textarea, [tabindex]:not([tabindex="-1"])');
  firstFocusable?.focus();

  // Escキーでモーダルを閉じる
  const handleEscape = (e) => {
    if (e.key === 'Escape') {
      closeModal();
    }
  };

  // フォーカストラップ
  const handleTab = (e) => {
    if (e.key !== 'Tab') return;

    const focusableElements = modal.querySelectorAll(
      'button:not([disabled]), [href], input, select, textarea, [tabindex]:not([tabindex="-1"])'
    );

    const firstElement = focusableElements[0];
    const lastElement = focusableElements[focusableElements.length - 1];

    // 最後の要素で Tab → 最初へ
    if (!e.shiftKey && document.activeElement === lastElement) {
      firstElement.focus();
      e.preventDefault();
    }

    // 最初の要素で Shift+Tab → 最後へ
    if (e.shiftKey && document.activeElement === firstElement) {
      lastElement.focus();
      e.preventDefault();
    }
  };

  modal.addEventListener('keydown', handleEscape);
  modal.addEventListener('keydown', handleTab);

  // クローズ処理
  function closeModal() {
    modal.style.display = 'none';
    modal.removeEventListener('keydown', handleEscape);
    modal.removeEventListener('keydown', handleTab);

    // 元の位置にフォーカスを戻す
    previousFocus?.focus();
  }

  return closeModal;
}
```

## 矢印キーナビゲーション

ツールバーやタブなどのカスタムウィジェット:

```javascript
class ArrowKeyNavigation {
  constructor(container, items) {
    this.container = container;
    this.items = items;
    this.currentIndex = 0;

    // 最初の要素のみ tabindex="0"
    this.items.forEach((item, index) => {
      item.setAttribute('tabindex', index === 0 ? '0' : '-1');
    });

    this.container.addEventListener('keydown', (e) => {
      this.handleKeyDown(e);
    });
  }

  handleKeyDown(e) {
    let newIndex = this.currentIndex;

    switch (e.key) {
      case 'ArrowRight':
      case 'ArrowDown':
        e.preventDefault();
        newIndex = (this.currentIndex + 1) % this.items.length;
        break;

      case 'ArrowLeft':
      case 'ArrowUp':
        e.preventDefault();
        newIndex = (this.currentIndex - 1 + this.items.length) % this.items.length;
        break;

      case 'Home':
        e.preventDefault();
        newIndex = 0;
        break;

      case 'End':
        e.preventDefault();
        newIndex = this.items.length - 1;
        break;

      default:
        return;
    }

    this.moveFocus(newIndex);
  }

  moveFocus(newIndex) {
    // 古い要素を tabindex="-1"
    this.items[this.currentIndex].setAttribute('tabindex', '-1');

    // 新しい要素を tabindex="0" にしてフォーカス
    this.items[newIndex].setAttribute('tabindex', '0');
    this.items[newIndex].focus();

    this.currentIndex = newIndex;
  }
}

// 使用例
const toolbar = document.querySelector('[role="toolbar"]');
const buttons = toolbar.querySelectorAll('button');
new ArrowKeyNavigation(toolbar, buttons);
```

## フォーカス管理（SPA）

ページ遷移後のフォーカス管理:

```javascript
// ページ遷移時
function navigateToPage(pageContent) {
  // コンテンツを更新
  document.getElementById('content').innerHTML = pageContent;

  // 新しいコンテンツの見出しにフォーカス
  const heading = document.querySelector('#content h1');
  if (heading) {
    heading.setAttribute('tabindex', '-1');
    heading.focus();

    // フォーカス後は tabindex を削除
    heading.addEventListener('blur', () => {
      heading.removeAttribute('tabindex');
    }, { once: true });
  }

  // スクリーンリーダーに通知
  const announcement = document.createElement('div');
  announcement.setAttribute('role', 'status');
  announcement.setAttribute('aria-live', 'polite');
  announcement.textContent = 'ページが読み込まれました';
  document.body.appendChild(announcement);

  setTimeout(() => {
    announcement.remove();
  }, 1000);
}
```

## まとめ

キーボードナビゲーションは、アクセシビリティの基礎です:

1. **すべてアクセス可能** - マウス不要で操作可能
2. **明確なフォーカス** - 現在位置が常にわかる
3. **論理的な順序** - 自然なTab順序
4. **標準的な操作** - 期待通りのキーボード操作
5. **実機テスト** - キーボードのみで動作確認

これらを実装することで、すべてのユーザーが快適にWebサイトを利用できます。
